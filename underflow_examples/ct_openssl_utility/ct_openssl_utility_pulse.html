53/*-
54 * The boolean methods return a bitmask of all ones (0xff...f) for true
55 * and 0 for false. This is useful for choosing a value based on the result
56 * of a conditional in constant time. For example,
57 *
58 * if (a < b) {
59 * c = a;
60 * } else {
61 * c = b;
62 * }
63 *
64 * can be written as
65 *
66 * unsigned int lt = constant_time_lt(a, b);
67 * c = constant_time_select(lt, a, b);
68 */
69
70/*
71 * Returns the given value with the MSB copied to all the other
72 * bits. Uses the fact that arithmetic shift shifts-in the sign bit.
73 * However, this is not ensured by the C standard so you may need to
74 * replace this with something else on odd CPUs.
75 */
76static unsigned int constant_time_msb(unsigned int a);
77
78/*
79 * Returns 0xff..f if a < b and 0 otherwise.
80 */
81static unsigned int constant_time_lt(unsigned int a, unsigned int b);
82/* Convenience method for getting an 8-bit mask. */
83static unsigned char constant_time_lt_8(unsigned int a,
84 unsigned int b);
85
86/*
87 * Returns 0xff..f if a >= b and 0 otherwise.
88 */
89static unsigned int constant_time_ge(unsigned int a, unsigned int b);
90/* Convenience method for getting an 8-bit mask. */
91static unsigned char constant_time_ge_8(unsigned int a,
92 unsigned int b);
93
94/*
95 * Returns 0xff..f if a == 0 and 0 otherwise.
96 */
97static unsigned int constant_time_is_zero(unsigned int a);
98/* Convenience method for getting an 8-bit mask. */
99static unsigned char constant_time_is_zero_8(unsigned int a);
100
101/*
102 * Returns 0xff..f if a == b and 0 otherwise.
103 */
104static unsigned int constant_time_eq(unsigned int a, unsigned int b);
105/* Convenience method for getting an 8-bit mask. */
106static unsigned char constant_time_eq_8(unsigned int a,
107 unsigned int b);
108/* Signed integers. */
109static unsigned int constant_time_eq_int(int a, int b);
110/* Convenience method for getting an 8-bit mask. */
111static unsigned char constant_time_eq_int_8(int a, int b);
112
113/*-
114 * Returns (mask & a) | (~mask & b).
115 *
116 * When |mask| is all 1s or all 0s (as returned by the methods above),
117 * the select methods return either |a| (if |mask| is nonzero) or |b|
118 * (if |mask| is zero).
119 */
120static unsigned int constant_time_select(unsigned int mask,
121 unsigned int a,
122 unsigned int b);
123/* Convenience method for unsigned chars. */
124static unsigned char constant_time_select_8(unsigned char mask,
125 unsigned char a,
126 unsigned char b);
127/* Convenience method for signed integers. */
128static int constant_time_select_int(unsigned int mask, int a, int b);
