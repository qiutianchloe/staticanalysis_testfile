58/*-
59 * The boolean methods return a bitmask of all ones (0xff...f) for true
60 * and 0 for false. This is useful for choosing a value based on the result
61 * of a conditional in constant time. For example,
62 *
63 * if (a < b) {
64 * c = a;
65 * } else {
66 * c = b;
67 * }
68 *
69 * can be written as
70 *
71 * unsigned int lt = constant_time_lt(a, b);
72 * c = constant_time_select(lt, a, b);
73 */
74
75/*
76 * Returns the given value with the MSB copied to all the other
77 * bits. Uses the fact that arithmetic shift shifts-in the sign bit.
78 * However, this is not ensured by the C standard so you may need to
79 * replace this with something else on odd CPUs.
80 */
81static unsigned int constant_time_msb(unsigned int a);
82
83/*
84 * Returns 0xff..f if a < b and 0 otherwise.
85 */
86static unsigned int constant_time_lt(unsigned int a, unsigned int b);
87/* Convenience method for getting an 8-bit mask. */
88static unsigned char constant_time_lt_8(unsigned int a,
89 unsigned int b);
90
91/*
92 * Returns 0xff..f if a >= b and 0 otherwise.
93 */
94static unsigned int constant_time_ge(unsigned int a, unsigned int b);
95/* Convenience method for getting an 8-bit mask. */
96static unsigned char constant_time_ge_8(unsigned int a,
97 unsigned int b);
98
99/*
100 * Returns 0xff..f if a == 0 and 0 otherwise.
101 */
102static unsigned int constant_time_is_zero(unsigned int a);
103/* Convenience method for getting an 8-bit mask. */
104static unsigned char constant_time_is_zero_8(unsigned int a);
105
106/*
107 * Returns 0xff..f if a == b and 0 otherwise.
108 */
109static unsigned int constant_time_eq(unsigned int a, unsigned int b);
110/* Convenience method for getting an 8-bit mask. */
111static unsigned char constant_time_eq_8(unsigned int a,
112 unsigned int b);
113/* Signed integers. */
114static unsigned int constant_time_eq_int(int a, int b);
115/* Convenience method for getting an 8-bit mask. */
116static unsigned char constant_time_eq_int_8(int a, int b);

118/*-
119 * Returns (mask & a) | (~mask & b).
120 *
121 * When |mask| is all 1s or all 0s (as returned by the methods above),
122 * the select methods return either |a| (if |mask| is nonzero) or |b|
123 * (if |mask| is zero).
124 */
125static unsigned int constant_time_select(unsigned int mask,
126 unsigned int a,
127 unsigned int b);
128/* Convenience method for unsigned chars. */
129static unsigned char constant_time_select_8(unsigned char mask,
130 unsigned char a,
131 unsigned char b);
132/* Convenience method for signed integers. */
133static int constant_time_select_int(unsigned int mask, int a, int b);
